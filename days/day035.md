# Day 035: 触发器 — Trigger

## 学习目标

- 理解触发器的概念及其在数据库中的作用
- 掌握 BEFORE/AFTER/INSTEAD OF 三种触发时机
- 区分行级触发器和语句级触发器的使用场景
- 熟练使用 NEW 和 OLD 引用操作前后的数据行
- 能够实现完整的审计日志功能

## 理论讲解

### 什么是触发器？

触发器（Trigger）是与表关联的数据库对象，在特定事件（INSERT、UPDATE、DELETE、TRUNCATE）发生时自动执行。触发器可以用来：

- **数据验证**：在插入或更新前检查数据合法性
- **自动填充**：自动设置 created_at、updated_at 等字段
- **审计日志**：记录数据的变更历史
- **级联操作**：数据变更时自动更新关联表
- **数据同步**：将变更同步到其他表或系统

### 触发器结构

创建触发器需要两步：

1. **创建触发器函数**：返回类型为 `TRIGGER` 的函数
2. **创建触发器**：将函数绑定到表和事件

```sql
-- 步骤1：创建触发器函数
CREATE FUNCTION trigger_func() RETURNS TRIGGER AS $$ ... $$ LANGUAGE plpgsql;

-- 步骤2：创建触发器
CREATE TRIGGER trigger_name
    {BEFORE | AFTER | INSTEAD OF}
    {INSERT | UPDATE | DELETE | TRUNCATE}
    ON table_name
    [FOR EACH {ROW | STATEMENT}]
    [WHEN (condition)]
    EXECUTE FUNCTION trigger_func();
```

### 触发时机

| 时机 | 说明 | 典型用途 |
|------|------|----------|
| `BEFORE` | 在操作执行前触发 | 数据验证、自动填充、修改数据 |
| `AFTER` | 在操作执行后触发 | 审计日志、通知、级联更新 |
| `INSTEAD OF` | 替代操作执行（仅用于视图） | 使视图可更新 |

### 行级 vs 语句级

| 级别 | 触发次数 | NEW/OLD | 典型用途 |
|------|----------|---------|----------|
| `FOR EACH ROW` | 每受影响的行触发一次 | 可用 | 数据验证、审计每行变更 |
| `FOR EACH STATEMENT` | 每条语句触发一次 | 不可用 | 批量操作后的汇总通知 |

### NEW 和 OLD 引用

| 操作 | NEW | OLD |
|------|-----|-----|
| INSERT | 包含新插入的行 | 不可用 |
| UPDATE | 包含更新后的行 | 包含更新前的行 |
| DELETE | 不可用 | 包含被删除的行 |

在 BEFORE 触发器中，可以修改 NEW 的值来改变实际写入的数据。

### 触发器函数返回值

- **BEFORE 行级触发器**：返回 NEW 表示继续操作，返回 NULL 表示取消操作
- **AFTER 行级触发器**：返回值被忽略
- **语句级触发器**：返回值被忽略

## 代码示例

### 示例 1：自动更新时间戳

```sql
-- 创建通用的 updated_at 触发器函数
CREATE OR REPLACE FUNCTION update_timestamp()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    -- 自动设置 updated_at 为当前时间
    NEW.updated_at := NOW();
    RETURN NEW;
END;
$$;

-- 创建示例表
CREATE TABLE articles (
    id         SERIAL PRIMARY KEY,
    title      VARCHAR(200) NOT NULL,
    content    TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 绑定触发器到表
CREATE TRIGGER trg_articles_update_timestamp
    BEFORE UPDATE ON articles
    FOR EACH ROW
    EXECUTE FUNCTION update_timestamp();

-- 测试
INSERT INTO articles (title, content) VALUES ('测试文章', '内容...');
-- 等待几秒后更新
UPDATE articles SET title = '修改后的标题' WHERE id = 1;
-- updated_at 会自动更新
SELECT id, title, created_at, updated_at FROM articles;
```

### 示例 2：数据验证触发器

```sql
-- 在插入或更新前验证数据
CREATE OR REPLACE FUNCTION validate_employee()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    -- 验证薪资范围
    IF NEW.salary < 0 THEN
        RAISE EXCEPTION '薪资不能为负数：%', NEW.salary;
    END IF;

    IF NEW.salary > 1000000 THEN
        RAISE EXCEPTION '薪资超出合理范围：%', NEW.salary;
    END IF;

    -- 验证邮箱格式
    IF NEW.email IS NOT NULL AND NEW.email !~ '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z]{2,}$' THEN
        RAISE EXCEPTION '邮箱格式不正确：%', NEW.email;
    END IF;

    -- 自动规范化：名字首字母大写
    NEW.first_name := INITCAP(TRIM(NEW.first_name));
    NEW.last_name  := INITCAP(TRIM(NEW.last_name));

    RETURN NEW;  -- 返回（可能已修改的）NEW 继续操作
END;
$$;

CREATE TRIGGER trg_validate_employee
    BEFORE INSERT OR UPDATE ON employees
    FOR EACH ROW
    EXECUTE FUNCTION validate_employee();

-- 测试：薪资为负会报错
INSERT INTO employees (first_name, last_name, salary)
VALUES ('john', 'doe', -100);  -- 报错

-- 测试：名字会自动变为首字母大写
INSERT INTO employees (first_name, last_name, salary)
VALUES ('john', 'doe', 5000);
-- first_name 自动变为 'John'，last_name 变为 'Doe'
```

### 示例 3：完整的审计日志实现

```sql
-- 创建审计日志表
CREATE TABLE audit_log (
    id           BIGSERIAL PRIMARY KEY,
    table_name   TEXT NOT NULL,
    operation    TEXT NOT NULL,       -- INSERT/UPDATE/DELETE
    record_id    TEXT,                -- 被操作记录的ID
    old_data     JSONB,              -- 变更前数据
    new_data     JSONB,              -- 变更后数据
    changed_by   TEXT DEFAULT CURRENT_USER,
    changed_at   TIMESTAMP DEFAULT NOW(),
    client_ip    INET
);

-- 创建通用审计触发器函数
CREATE OR REPLACE FUNCTION audit_trigger_func()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_old_data JSONB;
    v_new_data JSONB;
    v_record_id TEXT;
BEGIN
    -- 根据操作类型设置数据
    IF TG_OP = 'DELETE' THEN
        v_old_data := to_jsonb(OLD);
        v_new_data := NULL;
        v_record_id := OLD.id::TEXT;
    ELSIF TG_OP = 'UPDATE' THEN
        v_old_data := to_jsonb(OLD);
        v_new_data := to_jsonb(NEW);
        v_record_id := NEW.id::TEXT;
    ELSIF TG_OP = 'INSERT' THEN
        v_old_data := NULL;
        v_new_data := to_jsonb(NEW);
        v_record_id := NEW.id::TEXT;
    END IF;

    -- 插入审计记录
    INSERT INTO audit_log (table_name, operation, record_id, old_data, new_data, client_ip)
    VALUES (
        TG_TABLE_NAME,       -- 触发器所在表名
        TG_OP,               -- 操作类型
        v_record_id,
        v_old_data,
        v_new_data,
        inet_client_addr()   -- 客户端IP
    );

    -- AFTER 触发器返回值被忽略，但语法要求返回
    IF TG_OP = 'DELETE' THEN
        RETURN OLD;
    ELSE
        RETURN NEW;
    END IF;
END;
$$;

-- 为 articles 表启用审计
CREATE TRIGGER trg_articles_audit
    AFTER INSERT OR UPDATE OR DELETE ON articles
    FOR EACH ROW
    EXECUTE FUNCTION audit_trigger_func();

-- 测试审计功能
INSERT INTO articles (title, content) VALUES ('新文章', '内容A');
UPDATE articles SET content = '修改后的内容' WHERE id = 1;
DELETE FROM articles WHERE id = 1;

-- 查看审计日志
SELECT operation, record_id, old_data, new_data, changed_at
FROM audit_log
WHERE table_name = 'articles'
ORDER BY changed_at;
```

### 示例 4：条件触发器（WHEN 子句）

```sql
-- 仅在薪资变更时触发审计
CREATE OR REPLACE FUNCTION log_salary_change()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO salary_change_log (
        employee_id, old_salary, new_salary,
        change_percent, changed_at
    )
    VALUES (
        NEW.employee_id,
        OLD.salary,
        NEW.salary,
        ROUND((NEW.salary - OLD.salary) / OLD.salary * 100, 2),
        NOW()
    );

    -- 如果加薪幅度超过 50%，发出警告
    IF (NEW.salary - OLD.salary) / OLD.salary > 0.5 THEN
        RAISE WARNING '警告：员工 % 加薪幅度超过50%%（从 % 到 %）',
                      NEW.employee_id, OLD.salary, NEW.salary;
    END IF;

    RETURN NEW;
END;
$$;

-- 使用 WHEN 子句：仅在 salary 列实际发生变化时触发
CREATE TRIGGER trg_salary_change
    AFTER UPDATE ON employees
    FOR EACH ROW
    WHEN (OLD.salary IS DISTINCT FROM NEW.salary)
    EXECUTE FUNCTION log_salary_change();

-- 这次更新不会触发触发器（薪资未变）
UPDATE employees SET first_name = 'Alice' WHERE employee_id = 1;

-- 这次更新会触发触发器
UPDATE employees SET salary = 10000 WHERE employee_id = 1;
```

### 示例 5：INSTEAD OF 触发器（使视图可更新）

```sql
-- 创建一个跨表的视图
CREATE VIEW v_employee_dept AS
    SELECT e.employee_id, e.first_name, e.last_name,
           e.salary, d.department_name
    FROM employees e
    LEFT JOIN departments d ON e.department_id = d.department_id;

-- 创建 INSTEAD OF 触发器使视图支持 INSERT
CREATE OR REPLACE FUNCTION v_employee_dept_insert()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_dept_id INTEGER;
BEGIN
    -- 查找或创建部门
    SELECT department_id INTO v_dept_id
    FROM departments
    WHERE department_name = NEW.department_name;

    IF NOT FOUND THEN
        INSERT INTO departments (department_name)
        VALUES (NEW.department_name)
        RETURNING department_id INTO v_dept_id;
    END IF;

    -- 插入员工
    INSERT INTO employees (first_name, last_name, salary, department_id)
    VALUES (NEW.first_name, NEW.last_name, NEW.salary, v_dept_id);

    RETURN NEW;
END;
$$;

CREATE TRIGGER trg_v_employee_dept_insert
    INSTEAD OF INSERT ON v_employee_dept
    FOR EACH ROW
    EXECUTE FUNCTION v_employee_dept_insert();

-- 现在可以直接向视图插入数据
INSERT INTO v_employee_dept (first_name, last_name, salary, department_name)
VALUES ('张', '三', 8000, '技术部');
```

### 示例 6：语句级触发器

```sql
-- 语句级触发器：记录批量操作
CREATE OR REPLACE FUNCTION log_bulk_operation()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO operation_log (table_name, operation, executed_by, executed_at)
    VALUES (TG_TABLE_NAME, TG_OP, CURRENT_USER, NOW());

    RETURN NULL;  -- 语句级触发器返回值被忽略
END;
$$;

CREATE TRIGGER trg_employees_bulk_log
    AFTER INSERT OR UPDATE OR DELETE ON employees
    FOR EACH STATEMENT
    EXECUTE FUNCTION log_bulk_operation();

-- 即使影响100行，触发器也只执行一次
UPDATE employees SET salary = salary * 1.1 WHERE department_id = 10;
```

### 示例 7：触发器管理

```sql
-- 查看表上的所有触发器
SELECT trigger_name, event_manipulation, action_timing, action_orientation
FROM information_schema.triggers
WHERE event_object_table = 'articles';

-- 临时禁用触发器（需要超级用户或表所有者权限）
ALTER TABLE articles DISABLE TRIGGER trg_articles_audit;

-- 重新启用
ALTER TABLE articles ENABLE TRIGGER trg_articles_audit;

-- 禁用表上所有触发器
ALTER TABLE articles DISABLE TRIGGER ALL;

-- 启用所有触发器
ALTER TABLE articles ENABLE TRIGGER ALL;

-- 删除触发器
DROP TRIGGER IF EXISTS trg_articles_audit ON articles;

-- 删除触发器函数
DROP FUNCTION IF EXISTS audit_trigger_func();
```

### 练习 3：库存预警触发器

## 常见问题

### Q1：一个表上可以有多个触发器吗？执行顺序是什么？

可以有多个触发器。同类型触发器按名称的字母顺序执行。例如 `trg_a` 在 `trg_b` 之前执行。

完整的执行顺序：
1. BEFORE 语句级触发器
2. BEFORE 行级触发器（每行）
3. 实际的 INSERT/UPDATE/DELETE 操作
4. AFTER 行级触发器（每行）
5. AFTER 语句级触发器

### Q2：触发器会影响性能吗？

会。触发器增加了每次数据操作的开销。影响程度取决于：
- 触发器函数的复杂度
- 触发频率（行级 vs 语句级）
- 触发器中是否有额外的查询或写入操作

**建议：** 保持触发器函数简洁高效，避免在触发器中执行复杂查询。对于批量导入等场景，可以临时禁用触发器。

### Q3：TG_OP 等特殊变量还有哪些？

触发器函数中可用的特殊变量：

| 变量 | 类型 | 说明 |
|------|------|------|
| `NEW` | RECORD | 新行数据（INSERT/UPDATE） |
| `OLD` | RECORD | 旧行数据（UPDATE/DELETE） |
| `TG_NAME` | TEXT | 触发器名称 |
| `TG_WHEN` | TEXT | BEFORE/AFTER/INSTEAD OF |
| `TG_LEVEL` | TEXT | ROW/STATEMENT |
| `TG_OP` | TEXT | INSERT/UPDATE/DELETE/TRUNCATE |
| `TG_TABLE_NAME` | TEXT | 触发器所在表名 |
| `TG_TABLE_SCHEMA` | TEXT | 表所在的 schema |
| `TG_NARGS` | INTEGER | 传递给触发器函数的参数数量 |
| `TG_ARGV` | TEXT[] | 传递给触发器函数的参数数组 |
