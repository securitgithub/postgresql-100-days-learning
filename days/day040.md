```sql
-- CMS 数据库 DDL

-- 用户表
CREATE TABLE cms_users (
    id         BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    username   VARCHAR(50) NOT NULL UNIQUE,
    email      VARCHAR(255) NOT NULL UNIQUE,
    role       VARCHAR(20) NOT NULL DEFAULT 'author'
               CHECK (role IN ('admin', 'editor', 'author', 'reader')),
    status     VARCHAR(20) NOT NULL DEFAULT 'active'
               CHECK (status IN ('active', 'inactive', 'banned')),
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 分类表（支持树形结构）
CREATE TABLE cms_categories (
    id         BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    name       VARCHAR(100) NOT NULL,
    slug       VARCHAR(100) NOT NULL UNIQUE,
    parent_id  BIGINT REFERENCES cms_categories(id) ON DELETE SET NULL,
    sort_order INTEGER NOT NULL DEFAULT 0,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 标签表
CREATE TABLE cms_tags (
    id         BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    name       VARCHAR(50) NOT NULL UNIQUE,
    slug       VARCHAR(50) NOT NULL UNIQUE,
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 文章表（使用 JSONB 存储元数据）
CREATE TABLE cms_articles (
    id           BIGINT GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    title        VARCHAR(500) NOT NULL,
    slug         VARCHAR(500) NOT NULL UNIQUE,
    content      TEXT,
    excerpt      VARCHAR(1000),
    author_id    BIGINT NOT NULL REFERENCES cms_users(id),
    category_id  BIGINT REFERENCES cms_categories(id) ON DELETE SET NULL,
    status       VARCHAR(20) NOT NULL DEFAULT 'draft'
                 CHECK (status IN ('draft', 'review', 'published', 'archived')),
    metadata     JSONB NOT NULL DEFAULT '{}',
    -- metadata 示例结构:
    -- {
    --   "seo_title": "...",
    --   "seo_description": "...",
    --   "featured_image": "https://...",
    --   "reading_time_minutes": 5,
    --   "is_featured": false,
    --   "custom_fields": {"key": "value"}
    -- }
    view_count   INTEGER NOT NULL DEFAULT 0,
    published_at TIMESTAMPTZ,
    created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at   TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- 文章-标签关联表（多对多）
CREATE TABLE cms_article_tags (
    article_id BIGINT NOT NULL REFERENCES cms_articles(id) ON DELETE CASCADE,
    tag_id     BIGINT NOT NULL REFERENCES cms_tags(id) ON DELETE CASCADE,
    PRIMARY KEY (article_id, tag_id)
);

-- 审计日志表
CREATE TABLE cms_audit_log (
    id           BIGSERIAL PRIMARY KEY,
    table_name   VARCHAR(100) NOT NULL,
    record_id    BIGINT,
    operation    VARCHAR(10) NOT NULL,
    old_data     JSONB,
    new_data     JSONB,
    changed_by   VARCHAR(100) DEFAULT CURRENT_USER,
    changed_at   TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- ========================================
-- 索引规划
-- ========================================
CREATE INDEX idx_articles_author ON cms_articles(author_id);
CREATE INDEX idx_articles_category ON cms_articles(category_id);
CREATE INDEX idx_articles_status ON cms_articles(status);
CREATE INDEX idx_articles_published ON cms_articles(published_at DESC)
    WHERE status = 'published';
CREATE INDEX idx_articles_metadata ON cms_articles USING GIN(metadata);
CREATE INDEX idx_article_tags_tag ON cms_article_tags(tag_id);
CREATE INDEX idx_categories_parent ON cms_categories(parent_id);
CREATE INDEX idx_audit_log_table ON cms_audit_log(table_name, changed_at);

COMMENT ON TABLE cms_articles IS '文章表，metadata 存储 SEO 和自定义元数据';
COMMENT ON COLUMN cms_articles.metadata IS 'JSONB 元数据：seo_title, seo_description, featured_image, reading_time_minutes, is_featured, custom_fields';
```

### 示例 3：触发器 — 自动更新时间戳和审计日志

```sql
-- 通用时间戳更新触发器
CREATE OR REPLACE FUNCTION trg_update_timestamp()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    NEW.updated_at := NOW();
    RETURN NEW;
END;
$$;

CREATE TRIGGER trg_articles_timestamp
    BEFORE UPDATE ON cms_articles
    FOR EACH ROW
    EXECUTE FUNCTION trg_update_timestamp();

CREATE TRIGGER trg_users_timestamp
    BEFORE UPDATE ON cms_users
    FOR EACH ROW
    EXECUTE FUNCTION trg_update_timestamp();

-- 文章审计触发器
CREATE OR REPLACE FUNCTION trg_article_audit()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        INSERT INTO cms_audit_log (table_name, record_id, operation, new_data)
        VALUES ('cms_articles', NEW.id, 'INSERT', to_jsonb(NEW));
    ELSIF TG_OP = 'UPDATE' THEN
        INSERT INTO cms_audit_log (table_name, record_id, operation, old_data, new_data)
        VALUES ('cms_articles', NEW.id, 'UPDATE', to_jsonb(OLD), to_jsonb(NEW));
    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO cms_audit_log (table_name, record_id, operation, old_data)
        VALUES ('cms_articles', OLD.id, 'DELETE', to_jsonb(OLD));
    END IF;

    IF TG_OP = 'DELETE' THEN
        RETURN OLD;
    ELSE
        RETURN NEW;
    END IF;
END;
$$;

CREATE TRIGGER trg_articles_audit
    AFTER INSERT OR UPDATE OR DELETE ON cms_articles
    FOR EACH ROW
    EXECUTE FUNCTION trg_article_audit();

-- 文章发布时自动设置 published_at
CREATE OR REPLACE FUNCTION trg_article_publish()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    IF NEW.status = 'published' AND OLD.status != 'published' THEN
        NEW.published_at := NOW();
    END IF;
    RETURN NEW;
END;
$$;

CREATE TRIGGER trg_articles_publish
    BEFORE UPDATE ON cms_articles
    FOR EACH ROW
    WHEN (OLD.status IS DISTINCT FROM NEW.status)
    EXECUTE FUNCTION trg_article_publish();
```

### 示例 4：存储过程 — 文章发布流程

```sql
-- 发布文章的存储过程
CREATE OR REPLACE PROCEDURE publish_article(
    p_article_id BIGINT,
    p_editor_id  BIGINT,
    INOUT p_result TEXT DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_article    RECORD;
    v_editor     RECORD;
BEGIN
    -- 验证文章
    SELECT id, title, status, author_id
    INTO v_article
    FROM cms_articles
    WHERE id = p_article_id;

    IF NOT FOUND THEN
        p_result := '失败：文章不存在';
        RETURN;
    END IF;

    IF v_article.status = 'published' THEN
        p_result := '失败：文章已经是发布状态';
        RETURN;
    END IF;

    -- 验证编辑权限
    SELECT id, username, role
    INTO v_editor
    FROM cms_users
    WHERE id = p_editor_id AND status = 'active';

    IF NOT FOUND THEN
        p_result := '失败：编辑用户不存在或已停用';
        RETURN;
    END IF;

    IF v_editor.role NOT IN ('admin', 'editor') THEN
        p_result := '失败：用户无发布权限';
        RETURN;
    END IF;

    -- 更新文章状态
    UPDATE cms_articles
    SET status = 'published'
    WHERE id = p_article_id;

    p_result := format('成功：文章《%s》已由 %s 发布',
                       v_article.title, v_editor.username);
END;
$$;

-- 调用
CALL publish_article(5, 2, NULL);
```

### 示例 5：SQL 函数 — 文章查询

```sql
-- 获取文章完整信息（含标签数组）
CREATE OR REPLACE FUNCTION get_article_detail(p_article_id BIGINT)
RETURNS TABLE(
    id            BIGINT,
    title         VARCHAR,
    author        VARCHAR,
    category      VARCHAR,
    status        VARCHAR,
    tags          TEXT[],
    metadata      JSONB,
    view_count    INTEGER,
    published_at  TIMESTAMPTZ
)
LANGUAGE sql
STABLE
AS $$
    SELECT
        a.id, a.title,
        u.username AS author,
        c.name AS category,
        a.status,
        COALESCE(array_agg(t.name) FILTER (WHERE t.name IS NOT NULL), '{}') AS tags,
        a.metadata,
        a.view_count,
        a.published_at
    FROM cms_articles a
    JOIN cms_users u ON a.author_id = u.id
    LEFT JOIN cms_categories c ON a.category_id = c.id
    LEFT JOIN cms_article_tags at ON a.id = at.article_id
    LEFT JOIN cms_tags t ON at.tag_id = t.id
    WHERE a.id = p_article_id
    GROUP BY a.id, u.username, c.name;
$$;

-- 调用
SELECT * FROM get_article_detail(1);

-- 管理标签的函数
CREATE OR REPLACE FUNCTION set_article_tags(
    p_article_id BIGINT,
    p_tag_names  TEXT[]
)
RETURNS TEXT[]
LANGUAGE plpgsql
AS $$
DECLARE
    v_tag_name TEXT;
    v_tag_id   BIGINT;
    v_result   TEXT[] := '{}';
BEGIN
    -- 删除现有标签关联
    DELETE FROM cms_article_tags WHERE article_id = p_article_id;

    -- 添加新标签
    FOREACH v_tag_name IN ARRAY p_tag_names LOOP
        -- 查找或创建标签
        SELECT id INTO v_tag_id FROM cms_tags WHERE name = v_tag_name;

        IF NOT FOUND THEN
            INSERT INTO cms_tags (name, slug)
            VALUES (v_tag_name, LOWER(REPLACE(v_tag_name, ' ', '-')))
            RETURNING id INTO v_tag_id;
        END IF;

        -- 创建关联
        INSERT INTO cms_article_tags (article_id, tag_id)
        VALUES (p_article_id, v_tag_id)
        ON CONFLICT DO NOTHING;

        v_result := array_append(v_result, v_tag_name);
    END LOOP;

    RETURN v_result;
END;
$$;

-- 调用：为文章设置标签
SELECT set_article_tags(1, ARRAY['PostgreSQL', 'JSONB', '新标签']);
```

### 示例 6：窗口函数查询 — 数据分析

```sql
-- 文章排名：按浏览量排名（含并列处理）
SELECT
    title,
    view_count,
    RANK() OVER (ORDER BY view_count DESC) AS view_rank,
    DENSE_RANK() OVER (ORDER BY view_count DESC) AS dense_view_rank
FROM cms_articles
WHERE status = 'published';

-- 作者统计：每位作者的文章数量和排名
SELECT
    u.username,
    COUNT(a.id) AS article_count,
    SUM(a.view_count) AS total_views,
    RANK() OVER (ORDER BY SUM(a.view_count) DESC) AS author_rank,
    ROUND(AVG(a.view_count), 0) AS avg_views
FROM cms_users u
LEFT JOIN cms_articles a ON u.id = a.author_id AND a.status = 'published'
WHERE u.role IN ('author', 'editor', 'admin')
GROUP BY u.id, u.username
ORDER BY total_views DESC NULLS LAST;

-- 分类内排名：每个分类中文章的浏览量排名
SELECT
    c.name AS category,
    a.title,
    a.view_count,
    ROW_NUMBER() OVER (PARTITION BY c.id ORDER BY a.view_count DESC) AS rank_in_category,
    SUM(a.view_count) OVER (PARTITION BY c.id) AS category_total_views
FROM cms_articles a
JOIN cms_categories c ON a.category_id = c.id
WHERE a.status = 'published'
ORDER BY c.name, rank_in_category;

-- 发布趋势：按周统计发布数量和累计浏览量
SELECT
    DATE_TRUNC('week', published_at) AS week_start,
    COUNT(*) AS articles_published,
    SUM(view_count) AS week_views,
    SUM(COUNT(*)) OVER (ORDER BY DATE_TRUNC('week', published_at)) AS cumulative_articles,
    SUM(SUM(view_count)) OVER (ORDER BY DATE_TRUNC('week', published_at)) AS cumulative_views
FROM cms_articles
WHERE status = 'published'
GROUP BY DATE_TRUNC('week', published_at)
ORDER BY week_start;

-- 标签热度分析
SELECT
    t.name AS tag,
    COUNT(at.article_id) AS article_count,
    SUM(a.view_count) AS total_views,
    ROUND(AVG(a.view_count), 0) AS avg_views,
    RANK() OVER (ORDER BY SUM(a.view_count) DESC) AS hot_rank
FROM cms_tags t
JOIN cms_article_tags at ON t.id = at.tag_id
JOIN cms_articles a ON at.article_id = a.id AND a.status = 'published'
GROUP BY t.id, t.name
ORDER BY hot_rank;

-- JSONB 元数据分析：精选文章 vs 普通文章
SELECT
    CASE
        WHEN (metadata ->> 'is_featured')::BOOLEAN = TRUE THEN '精选文章'
        ELSE '普通文章'
    END AS article_type,
    COUNT(*) AS count,
    ROUND(AVG(view_count), 0) AS avg_views,
    SUM(view_count) AS total_views,
    ROUND(AVG((metadata ->> 'reading_time_minutes')::NUMERIC), 1) AS avg_reading_time
FROM cms_articles
WHERE status = 'published'
GROUP BY (metadata ->> 'is_featured')::BOOLEAN;
```

### 示例 7：综合查询 — CMS 仪表板数据

```sql
-- 仪表板汇总函数
CREATE OR REPLACE FUNCTION get_dashboard_stats()
RETURNS JSONB
LANGUAGE plpgsql
STABLE
AS $$
DECLARE
    v_stats JSONB;
BEGIN
    SELECT jsonb_build_object(
        'total_articles', (SELECT COUNT(*) FROM cms_articles),
        'published_articles', (SELECT COUNT(*) FROM cms_articles WHERE status = 'published'),
        'draft_articles', (SELECT COUNT(*) FROM cms_articles WHERE status = 'draft'),
        'total_views', (SELECT COALESCE(SUM(view_count), 0) FROM cms_articles),
        'total_users', (SELECT COUNT(*) FROM cms_users WHERE status = 'active'),
        'total_tags', (SELECT COUNT(*) FROM cms_tags),
        'total_categories', (SELECT COUNT(*) FROM cms_categories),
        'top_articles', (
            SELECT jsonb_agg(row_to_json(t))
            FROM (
                SELECT title, view_count, published_at
                FROM cms_articles
                WHERE status = 'published'
                ORDER BY view_count DESC
                LIMIT 5
            ) t
        ),
        'recent_activity', (
            SELECT jsonb_agg(row_to_json(t))
            FROM (
                SELECT table_name, operation, record_id, changed_at
                FROM cms_audit_log
                ORDER BY changed_at DESC
                LIMIT 10
            ) t
        )
    ) INTO v_stats;

    RETURN v_stats;
END;
$$;

-- 调用仪表板统计
SELECT jsonb_pretty(get_dashboard_stats());
```

## 常见问题

### Q1：第二阶段最重要的知识点是什么？

从实用角度排序：
1. **JSONB**：现代应用几乎必用，灵活存储半结构化数据
2. **触发器**：审计日志、自动填充是生产环境的标配
3. **PL/pgSQL 函数**：封装复杂业务逻辑的核心手段
4. **数据建模**：良好的表设计是一切的基础
5. **存储过程**：批量操作和事务控制的利器

### Q2：实际项目中函数和存储过程应该放多少业务逻辑？

这取决于团队和架构决策。两种主流思路：

**数据库中心（Thick Database）：**
- 大量业务逻辑在函数/存储过程中
- 优势：性能好、数据一致性高
- 劣势：难以版本控制、难以测试、与数据库耦合

**应用中心（Thin Database）：**
- 数据库仅负责存储和基本约束
- 业务逻辑在应用层
- 优势：易于测试、部署灵活、技术栈无关
- 劣势：可能有性能损失、数据一致性需应用层保证

**推荐折中方案：** 数据完整性逻辑（触发器、约束）放数据库，业务逻辑放应用层，复杂的数据分析查询用函数封装。

### Q3：如何将本节创建的数据库对象管理纳入版本控制？

推荐使用数据库迁移工具：

- **Flyway**：基于 SQL 文件的迁移
- **Liquibase**：支持 XML/YAML/SQL 格式
- **sqitch**：PostgreSQL 社区推荐的迁移工具
- **Alembic**：Python 项目常用（配合 SQLAlchemy）

基本原则：
- 每个变更对应一个迁移文件
- 迁移文件按时间或版本号排序
- 包含 up（升级）和 down（回滚）两个方向
- 将迁移文件纳入 Git 管理
