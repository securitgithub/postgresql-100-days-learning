## 理论讲解

### 1. SELECT 语句执行顺序

虽然 SELECT 是我们写在最前面的关键字，但 SQL 的实际执行顺序是：

```
1. FROM        -- 确定数据来源
2. WHERE       -- 行级过滤
3. GROUP BY    -- 分组
4. HAVING      -- 组级过滤
5. SELECT      -- 选择列、计算表达式
6. DISTINCT    -- 去重
7. ORDER BY    -- 排序
8. LIMIT/OFFSET -- 分页
```

理解执行顺序很重要，它解释了为什么不能在 WHERE 中使用列别名（因为 SELECT 在 WHERE 之后执行）。

### 2. 比较运算符

| 运算符 | 含义 | 示例 |
|--------|------|------|
| = | 等于 | `price = 100` |
| <> 或 != | 不等于 | `status <> 'deleted'` |
| < | 小于 | `age < 18` |
| > | 大于 | `price > 1000` |
| <= | 小于等于 | `stock <= 10` |
| >= | 大于等于 | `score >= 60` |

### 3. 逻辑运算符

| 运算符 | 含义 | 优先级 |
|--------|------|--------|
| NOT | 取反 | 最高 |
| AND | 与 | 中 |
| OR | 或 | 最低 |

### 4. NULL 的特殊性

NULL 不是一个"值"，而是"未知"。任何与 NULL 的比较都不返回 true：
- `NULL = NULL` → NULL（不是 true）
- `NULL <> 1` → NULL（不是 true）
- 只能使用 `IS NULL` 和 `IS NOT NULL` 来判断

### SELECT 基础语法

```sql
-- 查询所有列
SELECT * FROM employees;

-- 查询指定列
SELECT name, department, salary FROM employees;

-- 使用表达式
SELECT
    name,
    salary,
    salary * 12 AS annual_salary,    -- 年薪
    salary * 12 * 1.1 AS with_bonus  -- 含10%奖金的年薪
FROM employees;

-- DISTINCT 去重
SELECT DISTINCT department FROM employees;
SELECT DISTINCT city FROM employees ORDER BY city;

-- DISTINCT ON (PostgreSQL 特有)：每组取第一行
SELECT DISTINCT ON (department)
    department, name, salary
FROM employees
ORDER BY department, salary DESC;  -- 每个部门薪资最高的人
```

### 列别名

```sql
-- 使用 AS 设置别名
SELECT
    name AS 姓名,
    department AS 部门,
    salary AS 月薪,
    salary * 12 AS 年薪
FROM employees;

-- AS 可以省略（但不推荐，降低可读性）
SELECT name 姓名, department 部门 FROM employees;

-- 别名中包含空格或关键字时需要双引号
SELECT
    name AS "Employee Name",
    salary AS "Monthly Salary (RMB)"
FROM employees;

-- 表别名
SELECT e.name, e.department, e.salary
FROM employees AS e;

-- 表别名在自连接中特别有用
SELECT
    e.name AS employee,
    m.name AS manager
FROM employees e
LEFT JOIN employees m ON e.manager_id = m.id;
```

### WHERE 条件过滤

```sql
-- 等值条件
SELECT name, department, salary
FROM employees
WHERE department = '技术部';

-- 不等条件
SELECT name, department, salary
FROM employees
WHERE department <> '技术部';

-- 数值比较
SELECT name, salary FROM employees WHERE salary > 20000;
SELECT name, salary FROM employees WHERE salary <= 15000;

-- 日期比较
SELECT name, hire_date
FROM employees
WHERE hire_date >= '2021-01-01';

SELECT name, hire_date
FROM employees
WHERE hire_date < '2020-01-01';

-- 布尔条件
SELECT name, is_active FROM employees WHERE is_active;       -- 在职
SELECT name, is_active FROM employees WHERE NOT is_active;   -- 离职
```

### AND / OR / NOT 组合条件

```sql
-- AND：所有条件都必须满足
SELECT name, department, salary
FROM employees
WHERE department = '技术部' AND salary > 20000;

-- OR：满足任一条件即可
SELECT name, department, salary
FROM employees
WHERE department = '技术部' OR department = '市场部';

-- NOT：取反
SELECT name, department
FROM employees
WHERE NOT department = '技术部';

-- 复合条件（注意优先级：NOT > AND > OR）
-- 查找：技术部且薪资 > 20000，或者市场部
SELECT name, department, salary
FROM employees
WHERE (department = '技术部' AND salary > 20000)
   OR department = '市场部';

-- 使用括号明确优先级（强烈推荐，避免歧义）
SELECT name, department, salary, city
FROM employees
WHERE department = '技术部'
  AND (city = '北京' OR city = '杭州')
  AND salary > 15000;
```

### LIKE / ILIKE 模式匹配

```sql
-- LIKE 模式匹配（区分大小写）
-- % 匹配任意长度字符（包括空字符）
-- _ 匹配恰好一个字符

-- 以"张"开头的名字
SELECT name FROM employees WHERE name LIKE '张%';

-- 以"工程师"结尾的职位
SELECT name, position FROM employees WHERE position LIKE '%工程师';

-- 包含"市场"的部门
SELECT name, department FROM employees WHERE department LIKE '%市场%';

-- 名字恰好两个字的员工（_ 匹配单个字符）
SELECT name FROM employees WHERE name LIKE '__';

-- NOT LIKE
SELECT name, position FROM employees WHERE position NOT LIKE '%工程师';

-- ILIKE：不区分大小写（PostgreSQL 特有）
SELECT name, email FROM employees WHERE email ILIKE '%COMPANY%';  -- 不区分大小写
SELECT name, email FROM employees WHERE email LIKE '%COMPANY%';   -- 区分大小写，无结果

-- 转义特殊字符
-- 如果要匹配 % 或 _ 本身，使用 ESCAPE
SELECT * FROM employees WHERE email LIKE '%\_%' ESCAPE '\';  -- 查找包含下划线的邮箱
```

### IN / NOT IN

```sql
-- IN：值在列表中
SELECT name, department, salary
FROM employees
WHERE department IN ('技术部', '市场部');

-- 等价于 OR
SELECT name, department, salary
FROM employees
WHERE department = '技术部' OR department = '市场部';

-- NOT IN
SELECT name, city
FROM employees
WHERE city NOT IN ('北京', '上海');

-- IN 与子查询（预告，后续会详细学习）
SELECT name, salary
FROM employees
WHERE department IN (
    SELECT department FROM employees
    WHERE salary > 30000
);

-- 注意：NOT IN 与 NULL 的陷阱
-- 如果 IN 列表中包含 NULL，NOT IN 可能返回空结果
SELECT * FROM employees
WHERE manager_id NOT IN (1, 2, NULL);  -- 不会返回任何行！
-- 因为 NOT (manager_id = NULL) 始终不为 true

-- 安全的写法：过滤掉 NULL
SELECT * FROM employees
WHERE manager_id NOT IN (
    SELECT id FROM employees WHERE id IS NOT NULL
);
```

### BETWEEN

```sql
-- BETWEEN：范围查询（包含两端边界值）
SELECT name, salary
FROM employees
WHERE salary BETWEEN 15000 AND 25000;

-- 等价于
SELECT name, salary
FROM employees
WHERE salary >= 15000 AND salary <= 25000;

-- NOT BETWEEN
SELECT name, salary
FROM employees
WHERE salary NOT BETWEEN 15000 AND 25000;

-- 日期范围
SELECT name, hire_date
FROM employees
WHERE hire_date BETWEEN '2020-01-01' AND '2021-12-31';

-- 注意：BETWEEN 包含边界值
-- 如果只想要 2020 和 2021 年入职的员工，上面的写法正确
-- 但如果用 timestamp，要小心边界：
-- WHERE created_at BETWEEN '2020-01-01' AND '2020-12-31'
-- 不会包含 2020-12-31 23:59:59，应该写成：
-- WHERE created_at >= '2020-01-01' AND created_at < '2021-01-01'
```

### IS NULL / IS NOT NULL

```sql
-- IS NULL：查找 NULL 值
SELECT name, phone
FROM employees
WHERE phone IS NULL;

SELECT name, email
FROM employees
WHERE email IS NULL;

-- IS NOT NULL
SELECT name, phone
FROM employees
WHERE phone IS NOT NULL;

-- 常见错误：不要用 = NULL
SELECT name FROM employees WHERE phone = NULL;      -- 错误！永远不返回行
SELECT name FROM employees WHERE phone IS NULL;      -- 正确！

SELECT name FROM employees WHERE phone <> NULL;      -- 错误！永远不返回行
SELECT name FROM employees WHERE phone IS NOT NULL;  -- 正确！

-- COALESCE：为 NULL 提供默认值
SELECT
    name,
    COALESCE(phone, '未填写') AS phone,
    COALESCE(email, '无邮箱') AS email
FROM employees;

-- NULLIF：当两个值相等时返回 NULL（常用于避免除零错误）
SELECT NULLIF(10, 10);  -- 返回 NULL
SELECT NULLIF(10, 20);  -- 返回 10
-- 示例：安全除法
-- SELECT total / NULLIF(count, 0) FROM stats;
```

### 综合查询练习

```sql
-- 查询1：技术部中，月薪超过 20000 且在北京或杭州工作的在职员工
SELECT name, position, salary, city
FROM employees
WHERE department = '技术部'
  AND salary > 20000
  AND city IN ('北京', '杭州')
  AND is_active = true
ORDER BY salary DESC;

-- 查询2：2020年及以后入职的、薪资在 10000 到 20000 之间的员工
SELECT name, department, salary, hire_date
FROM employees
WHERE hire_date >= '2020-01-01'
  AND salary BETWEEN 10000 AND 20000
ORDER BY hire_date;

-- 查询3：找出没有经理（顶级管理者）且薪资最高的 3 人
SELECT name, department, position, salary
FROM employees
WHERE manager_id IS NULL
ORDER BY salary DESC
LIMIT 3;

-- 查询4：找出姓名包含"工程师"职位的员工，按部门和薪资排序
SELECT name, department, position, salary
FROM employees
WHERE position LIKE '%工程师'
ORDER BY department, salary DESC;

-- 查询5：薪资统计概览（使用表达式）
SELECT
    name,
    salary AS 月薪,
    salary * 12 AS 年薪,
    CASE
        WHEN salary >= 30000 THEN '高薪'
        WHEN salary >= 20000 THEN '中高'
        WHEN salary >= 15000 THEN '中等'
        ELSE '标准'
    END AS 薪资等级,
    CASE
        WHEN hire_date < '2019-01-01' THEN '资深'
        WHEN hire_date < '2021-01-01' THEN '中期'
        ELSE '新人'
    END AS 资历等级
FROM employees
WHERE is_active
ORDER BY salary DESC;

-- 查询6：使用 CASE 做条件统计
SELECT
    department,
    count(*) AS 总人数,
    count(*) FILTER (WHERE salary >= 20000) AS 高薪人数,
    count(*) FILTER (WHERE salary < 15000) AS 低薪人数
FROM employees
WHERE is_active
GROUP BY department
ORDER BY 总人数 DESC;
```

## 常见问题

### Q1: 为什么不能在 WHERE 中使用列别名？

**A:** 因为 SQL 的执行顺序是 FROM → WHERE → SELECT。别名在 SELECT 阶段才定义，WHERE 执行时别名还不存在。例如：
```sql
-- 错误：WHERE 中不能用别名
SELECT salary * 12 AS annual_salary FROM employees WHERE annual_salary > 200000;

-- 正确方式1：重复表达式
SELECT salary * 12 AS annual_salary FROM employees WHERE salary * 12 > 200000;

-- 正确方式2：使用子查询
SELECT * FROM (
    SELECT name, salary * 12 AS annual_salary FROM employees
) sub WHERE annual_salary > 200000;
```

### Q2: LIKE 和正则表达式（~）有什么区别？

**A:**
- `LIKE`：简单的通配符匹配，只有 `%` 和 `_` 两个通配符，适合简单的前缀/后缀/包含匹配
- `~`（正则表达式）：强大的模式匹配，支持完整的正则语法
- `ILIKE`（PostgreSQL 特有）：不区分大小写的 LIKE
- `~*`：不区分大小写的正则匹配
- 性能上 LIKE 通常更快，尤其是前缀匹配（`LIKE 'abc%'`）可以利用 B-tree 索引

### Q3: LIMIT 和 OFFSET 的性能问题？

**A:** `LIMIT` 本身没有性能问题，但 `OFFSET` 在值很大时会很慢，因为数据库需要先计算并跳过前 N 行。对于大表的分页查询，推荐使用"键集分页"（Keyset Pagination）代替 OFFSET：
```sql
-- 慢：OFFSET 大时性能差
SELECT * FROM employees ORDER BY id LIMIT 10 OFFSET 100000;

-- 快：基于上一页最后一条记录的 id
SELECT * FROM employees WHERE id > last_seen_id ORDER BY id LIMIT 10;
```
