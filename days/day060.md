### 2. 生产环境配置原则

- **安全第一**：最小权限原则，强制加密连接
- **性能合理**：根据硬件资源合理分配，避免过度配置
- **可观测性**：全面的日志记录和监控
- **高可用**：完善的备份策略和恢复能力
- **可维护性**：标准化配置，文档化操作流程

- CPU：16 核

### 示例1：核心性能参数配置

```sql
-- ============================================
-- 内存参数（基于 64GB 内存的服务器）
-- ============================================

-- 共享缓冲区：内存的 25%
ALTER SYSTEM SET shared_buffers = '16GB';

-- 有效缓存大小：内存的 75%（告诉优化器可用缓存量）
ALTER SYSTEM SET effective_cache_size = '48GB';

-- 工作内存：每个排序/哈希操作的内存
-- 注意：会话数 × work_mem 不应超过可用内存
ALTER SYSTEM SET work_mem = '64MB';

-- 维护操作内存
ALTER SYSTEM SET maintenance_work_mem = '2GB';

-- 大页面（需要 OS 配置支持）
ALTER SYSTEM SET huge_pages = 'try';

-- ============================================
-- WAL 与检查点配置
-- ============================================

ALTER SYSTEM SET wal_buffers = '64MB';
ALTER SYSTEM SET wal_level = 'replica';          -- 支持复制和归档
ALTER SYSTEM SET max_wal_size = '4GB';
ALTER SYSTEM SET min_wal_size = '1GB';
ALTER SYSTEM SET checkpoint_timeout = '15min';
ALTER SYSTEM SET checkpoint_completion_target = '0.9';
ALTER SYSTEM SET wal_compression = 'on';

-- ============================================
-- 连接配置
-- ============================================

ALTER SYSTEM SET max_connections = '200';
ALTER SYSTEM SET superuser_reserved_connections = '5';

-- ============================================
-- 后台进程配置
-- ============================================

ALTER SYSTEM SET effective_io_concurrency = '200';     -- SSD 推荐
ALTER SYSTEM SET random_page_cost = '1.1';             -- SSD 推荐
ALTER SYSTEM SET max_worker_processes = '16';
ALTER SYSTEM SET max_parallel_workers_per_gather = '4';
ALTER SYSTEM SET max_parallel_workers = '16';
ALTER SYSTEM SET max_parallel_maintenance_workers = '4';

-- ============================================
-- Autovacuum 配置
```

```bash
# 拒绝所有其他连接
```

### 示例3：用户与权限体系

```sql
-- ============================================
-- 角色体系创建
-- ============================================

-- DBA 角色
CREATE ROLE dba_role NOLOGIN;
GRANT pg_monitor TO dba_role;

-- 应用读写角色
CREATE ROLE app_readwrite NOLOGIN;

-- 只读角色
CREATE ROLE app_readonly NOLOGIN;

-- 创建登录用户
CREATE USER dba_admin WITH PASSWORD 'DBA_Str0ng_P@ss!' IN ROLE dba_role;
CREATE USER app_user WITH PASSWORD 'App_Str0ng_P@ss!' IN ROLE app_readwrite;
CREATE USER readonly WITH PASSWORD 'Ro_Str0ng_P@ss!' IN ROLE app_readonly;
CREATE USER monitor WITH PASSWORD 'Mon_Str0ng_P@ss!';
GRANT pg_monitor TO monitor;

-- 复制用户
CREATE USER repl_user WITH REPLICATION PASSWORD 'Repl_Str0ng_P@ss!';

-- ============================================
-- 数据库和 Schema 权限
-- ============================================

-- 创建应用数据库
CREATE DATABASE myapp_db OWNER dba_admin
    ENCODING 'UTF8'
    LC_COLLATE 'en_US.UTF-8'
    LC_CTYPE 'en_US.UTF-8';

-- 连接到应用数据库
\c myapp_db

-- 撤销 public schema 的默认权限
REVOKE ALL ON SCHEMA public FROM PUBLIC;
REVOKE CREATE ON SCHEMA public FROM PUBLIC;

-- 创建应用 schema
CREATE SCHEMA app AUTHORIZATION dba_admin;

-- 授权
GRANT USAGE ON SCHEMA app TO app_readwrite, app_readonly;
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA app TO app_readwrite;
GRANT SELECT ON ALL TABLES IN SCHEMA app TO app_readonly;
GRANT USAGE ON ALL SEQUENCES IN SCHEMA app TO app_readwrite;

-- 设置默认权限（新建对象自动授权）
ALTER DEFAULT PRIVILEGES IN SCHEMA app
    GRANT SELECT, INSERT, UPDATE, DELETE ON TABLES TO app_readwrite;
ALTER DEFAULT PRIVILEGES IN SCHEMA app
    GRANT SELECT ON TABLES TO app_readonly;
ALTER DEFAULT PRIVILEGES IN SCHEMA app
    GRANT USAGE ON SEQUENCES TO app_readwrite;

-- 设置搜索路径
ALTER DATABASE myapp_db SET search_path = app, public;
```

### 示例6：备份与归档配置

```sql
-- ============================================
-- WAL 归档配置
-- ============================================

ALTER SYSTEM SET archive_mode = 'on';
ALTER SYSTEM SET archive_command = 'test ! -f /archive/wal/%f && cp %p /archive/wal/%f';
ALTER SYSTEM SET archive_timeout = '300';  -- 5 分钟强制归档
```

```bash
# 创建归档目录
sudo mkdir -p /archive/wal /archive/basebackup
sudo chown -R postgres:postgres /archive

# 基础备份脚本
cat > /usr/local/bin/pg_backup.sh << 'SCRIPT'
#!/bin/bash
set -euo pipefail

BACKUP_DIR="/archive/basebackup"
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_PATH="${BACKUP_DIR}/base_${DATE}"
RETENTION_DAYS=7
LOG="/var/log/pg_backup.log"

echo "[$(date)] Starting backup..." >> "$LOG"

# 执行基础备份
pg_basebackup -U repl_user -D "$BACKUP_PATH" \
    -Ft -z -Xs -P \
    --checkpoint=fast \
    --label="base_${DATE}" \
    2>> "$LOG"

echo "[$(date)] Backup completed: $BACKUP_PATH" >> "$LOG"

# 清理旧备份
find "$BACKUP_DIR" -maxdepth 1 -name "base_*" -mtime +${RETENTION_DAYS} -exec rm -rf {} \;
echo "[$(date)] Old backups cleaned (>${RETENTION_DAYS} days)" >> "$LOG"

# 清理旧 WAL 归档
# 使用 pg_archivecleanup 保留最新备份之后的 WAL
OLDEST_BACKUP=$(ls -1d ${BACKUP_DIR}/base_* 2>/dev/null | head -1)
if [ -n "$OLDEST_BACKUP" ]; then
    BACKUP_LABEL=$(basename "$OLDEST_BACKUP")
    echo "[$(date)] Retaining WAL from: $BACKUP_LABEL" >> "$LOG"
fi
SCRIPT

chmod +x /usr/local/bin/pg_backup.sh

# 添加 crontab（每天凌晨 2 点备份）
echo "0 2 * * * postgres /usr/local/bin/pg_backup.sh" | sudo tee /etc/cron.d/pg_backup
```

### 示例8：综合健康检查脚本

```sql
-- ============================================
-- 生产环境健康检查
-- ============================================

-- 1. 基础信息
SELECT '=== 基础信息 ===' AS section;
SELECT version() AS pg_version;
SELECT pg_postmaster_start_time() AS server_start_time;
SELECT now() - pg_postmaster_start_time() AS uptime;
SELECT current_setting('data_directory') AS data_dir;

-- 2. 连接状态
SELECT '=== 连接状态 ===' AS section;
SELECT state, count(*) AS count
FROM pg_stat_activity
GROUP BY state;

SELECT current_setting('max_connections')::int AS max_conn,
       count(*) AS current_conn,
       round(100.0 * count(*) / current_setting('max_connections')::int, 1) AS usage_pct
FROM pg_stat_activity;

-- 3. 数据库大小
SELECT '=== 数据库大小 ===' AS section;
SELECT datname, pg_size_pretty(pg_database_size(datname)) AS size
FROM pg_database
WHERE datname NOT LIKE 'template%'
ORDER BY pg_database_size(datname) DESC;

-- 4. 缓存命中率
SELECT '=== 缓存命中率 ===' AS section;
SELECT datname,
       round(100.0 * blks_hit / NULLIF(blks_hit + blks_read, 0), 2) AS cache_hit_pct
FROM pg_stat_database
WHERE datname NOT LIKE 'template%';

-- 5. 事务统计
SELECT '=== 事务统计 ===' AS section;
SELECT datname,
       xact_commit, xact_rollback,
       round(100.0 * xact_rollback / NULLIF(xact_commit + xact_rollback, 0), 2) AS rollback_pct
FROM pg_stat_database
WHERE datname NOT LIKE 'template%';

-- 6. 长事务检查
SELECT '=== 长事务（> 5 分钟）===' AS section;
SELECT pid, usename, datname,
       now() - xact_start AS tx_duration,
       state, left(query, 60) AS query
FROM pg_stat_activity
WHERE xact_start IS NOT NULL
  AND now() - xact_start > interval '5 minutes'
  AND pid != pg_backend_pid();

-- 7. 锁等待
SELECT '=== 锁等待 ===' AS section;
SELECT pid, usename,
       pg_blocking_pids(pid) AS blocked_by,
       left(query, 60) AS query
FROM pg_stat_activity
WHERE cardinality(pg_blocking_pids(pid)) > 0;

-- 8. 复制状态（如有）
SELECT '=== 复制状态 ===' AS section;
SELECT client_addr, state, sent_lsn, write_lsn, flush_lsn, replay_lsn,
       now() - write_lag AS write_lag
FROM pg_stat_replication;

-- 9. 表膨胀 TOP5
SELECT '=== 表膨胀 TOP5 ===' AS section;
SELECT schemaname || '.' || relname AS table_name,
       n_dead_tup,
       round(100.0 * n_dead_tup / NULLIF(n_live_tup + n_dead_tup, 0), 1) AS dead_pct,
       pg_size_pretty(pg_table_size(relid)) AS size,
       last_autovacuum
FROM pg_stat_user_tables
ORDER BY n_dead_tup DESC
LIMIT 5;

-- 10. 备份与归档检查
SELECT '=== 归档状态 ===' AS section;
SELECT current_setting('archive_mode') AS archive_mode,
       current_setting('archive_command') AS archive_command;

SELECT archived_count, last_archived_wal, last_archived_time,
       failed_count, last_failed_wal, last_failed_time
FROM pg_stat_archiver;
```

## 常见问题

### Q1：生产环境最容易遗漏的配置是什么？

最常被遗漏的关键配置包括：(1) `idle_in_transaction_session_timeout`——不设置会导致长事务占用资源和锁；(2) `log_lock_waits`——不记录会让锁问题难以诊断；(3) `archive_mode`——不启用则无法做 PITR；(4) 默认权限——不收紧 public schema 权限可能导致安全隐患。

### Q2：如何评估配置参数是否合理？

关键指标：缓存命中率应 > 99%；`buffers_backend`（后端直接写出）应远小于 `buffers_checkpoint`；连接使用率不应超过 80%；autovacuum 不应频繁被取消。定期使用 `pg_stat_statements` 分析慢查询，用 pgBadger 分析日志趋势。

### Q3：一台服务器可以运行多少连接？

`max_connections` 不是越大越好。每个连接消耗约 10MB 内存。200 个连接的服务器需要约 2GB 仅用于连接开销。建议使用连接池（如 PgBouncer）处理高并发，让实际数据库连接保持在 200-500 之间。公式参考：max_connections = CPU 核数 × 4 作为起点。
