## 理论讲解

### 异常处理（EXCEPTION WHEN）

PL/pgSQL 通过 `EXCEPTION WHEN` 子句捕获和处理运行时错误。异常处理块位于 `BEGIN...END` 结构中：

```
BEGIN
    -- 可能出错的语句
EXCEPTION
    WHEN condition_name THEN
        -- 处理逻辑
    WHEN OTHERS THEN
        -- 捕获所有其他异常
END;
```

**重要特性：** 当异常被捕获时，该 BEGIN 块中的所有数据库操作会被自动回滚（内部使用 SAVEPOINT 机制），但变量值不会回滚。

常见异常条件：

| 异常名称 | 说明 |
|----------|------|
| `NO_DATA_FOUND` | SELECT INTO STRICT 未返回行 |
| `TOO_MANY_ROWS` | SELECT INTO STRICT 返回多行 |
| `UNIQUE_VIOLATION` | 违反唯一约束 |
| `FOREIGN_KEY_VIOLATION` | 违反外键约束 |
| `CHECK_VIOLATION` | 违反检查约束 |
| `NOT_NULL_VIOLATION` | 违反非空约束 |
| `DIVISION_BY_ZERO` | 除以零 |
| `OTHERS` | 匹配所有未被前面条件捕获的异常 |

### 游标（Cursor）

游标允许逐行处理查询结果，适用于需要逐条处理大量数据的场景。PL/pgSQL 支持两种游标：

1. **绑定游标（Bound Cursor）**：声明时就绑定了查询语句
2. **未绑定游标（Unbound Cursor）**：声明为 `REFCURSOR` 类型，运行时才打开查询

游标操作流程：声明 → 打开 → 获取（FETCH） → 关闭

### RETURN QUERY 与 RETURN NEXT

对于 `RETURNS SETOF` 函数，有两种方式返回多行数据：

- **RETURN QUERY**：一次性返回整个查询结果
- **RETURN NEXT**：逐行累积结果，最终一起返回

### PERFORM 与 EXECUTE

- **PERFORM**：执行不需要返回值的 SQL 语句（替代 SELECT 但丢弃结果）
- **EXECUTE**：执行动态构建的 SQL 字符串，支持 `USING` 参数绑定

## 代码示例

### 示例 1：基本异常处理

```sql
-- 安全获取员工薪资，包含异常处理
CREATE OR REPLACE FUNCTION safe_get_salary(p_emp_id INTEGER)
RETURNS NUMERIC
LANGUAGE plpgsql
STABLE
AS $$
DECLARE
    v_salary NUMERIC;
BEGIN
    -- 使用 STRICT 确保恰好返回一行
    SELECT salary INTO STRICT v_salary
    FROM employees
    WHERE employee_id = p_emp_id;

    RETURN v_salary;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RAISE NOTICE '员工 % 不存在', p_emp_id;
        RETURN NULL;
    WHEN TOO_MANY_ROWS THEN
        RAISE NOTICE '员工 % 存在重复记录', p_emp_id;
        RETURN NULL;
END;
$$;

SELECT safe_get_salary(101);  -- 正常返回薪资
SELECT safe_get_salary(999);  -- 返回 NULL，打印提示
```

### 示例 2：唯一约束冲突处理（UPSERT 模式）

```sql
-- 插入或更新用户信息
CREATE OR REPLACE FUNCTION upsert_user(
    p_email TEXT,
    p_name  TEXT,
    p_phone TEXT
)
RETURNS TEXT
LANGUAGE plpgsql
VOLATILE
AS $$
DECLARE
    v_action TEXT;
BEGIN
    -- 尝试插入
    INSERT INTO users (email, name, phone)
    VALUES (p_email, p_name, p_phone);
    v_action := '新增';

    RETURN format('用户 %s 已%s', p_email, v_action);

EXCEPTION
    WHEN UNIQUE_VIOLATION THEN
        -- 插入失败，改为更新
        UPDATE users
        SET name = p_name, phone = p_phone, updated_at = NOW()
        WHERE email = p_email;
        v_action := '更新';

        RETURN format('用户 %s 已%s', p_email, v_action);
END;
$$;

-- 第一次调用：插入
SELECT upsert_user('alice@test.com', 'Alice', '13800001111');
-- 第二次调用：更新
SELECT upsert_user('alice@test.com', 'Alice Wang', '13800002222');
```

### 示例 3：获取异常详细信息

```sql
-- 使用 GET STACKED DIAGNOSTICS 获取异常详情
CREATE OR REPLACE FUNCTION safe_divide(a NUMERIC, b NUMERIC)
RETURNS NUMERIC
LANGUAGE plpgsql
IMMUTABLE
AS $$
DECLARE
    v_result     NUMERIC;
    v_sqlstate   TEXT;
    v_message    TEXT;
    v_detail     TEXT;
    v_context    TEXT;
BEGIN
    v_result := a / b;
    RETURN v_result;

EXCEPTION
    WHEN OTHERS THEN
        -- 获取异常的详细诊断信息
        GET STACKED DIAGNOSTICS
            v_sqlstate = RETURNED_SQLSTATE,
            v_message  = MESSAGE_TEXT,
            v_detail   = PG_EXCEPTION_DETAIL,
            v_context  = PG_EXCEPTION_CONTEXT;

        RAISE NOTICE '错误代码: %', v_sqlstate;
        RAISE NOTICE '错误信息: %', v_message;
        RAISE NOTICE '错误详情: %', v_detail;
        RAISE NOTICE '错误上下文: %', v_context;

        RETURN NULL;
END;
$$;

SELECT safe_divide(10, 0);  -- 返回 NULL，并打印除零错误信息
```

### 示例 4：绑定游标

```sql
-- 使用绑定游标逐行处理高薪员工
CREATE OR REPLACE FUNCTION process_high_salary_employees(p_threshold NUMERIC)
RETURNS TEXT
LANGUAGE plpgsql
AS $$
DECLARE
    -- 声明绑定游标
    cur_emp CURSOR FOR
        SELECT employee_id, first_name, last_name, salary
        FROM employees
        WHERE salary > p_threshold
        ORDER BY salary DESC;

    v_rec    RECORD;
    v_report TEXT := '';
    v_count  INTEGER := 0;
BEGIN
    -- 打开游标
    OPEN cur_emp;

    LOOP
        -- 获取下一行
        FETCH cur_emp INTO v_rec;
        EXIT WHEN NOT FOUND;

        v_count := v_count + 1;
        v_report := v_report || format('%s. %s %s - ¥%s',
                    v_count, v_rec.first_name, v_rec.last_name,
                    v_rec.salary) || E'\n';
    END LOOP;

    -- 关闭游标
    CLOSE cur_emp;

    IF v_count = 0 THEN
        RETURN '没有找到薪资超过 ' || p_threshold || ' 的员工';
    END IF;

    RETURN format('共 %s 名高薪员工：', v_count) || E'\n' || v_report;
END;
$$;

SELECT process_high_salary_employees(10000);
```

### 示例 5：参数化游标

```sql
-- 带参数的游标
CREATE OR REPLACE FUNCTION dept_salary_cursor(p_dept_id INTEGER)
RETURNS TABLE(emp_name TEXT, salary NUMERIC, salary_level TEXT)
LANGUAGE plpgsql
STABLE
AS $$
DECLARE
    -- 参数化游标
    cur_sal CURSOR(c_dept INTEGER) FOR
        SELECT first_name || ' ' || last_name AS full_name, e.salary
        FROM employees e
        WHERE e.department_id = c_dept;

    v_rec RECORD;
BEGIN
    -- 打开时传参
    OPEN cur_sal(p_dept_id);

    LOOP
        FETCH cur_sal INTO v_rec;
        EXIT WHEN NOT FOUND;

        -- 使用 RETURN NEXT 逐行返回
        emp_name := v_rec.full_name;
        salary := v_rec.salary;

        IF v_rec.salary >= 10000 THEN
            salary_level := '高';
        ELSIF v_rec.salary >= 5000 THEN
            salary_level := '中';
        ELSE
            salary_level := '低';
        END IF;

        RETURN NEXT;
    END LOOP;

    CLOSE cur_sal;
END;
$$;

SELECT * FROM dept_salary_cursor(10);
```

### 示例 6：RETURN QUERY 返回集合

```sql
-- 使用 RETURN QUERY 返回集合结果
CREATE OR REPLACE FUNCTION search_employees(
    p_keyword TEXT DEFAULT NULL,
    p_dept_id INTEGER DEFAULT NULL,
    p_min_sal NUMERIC DEFAULT NULL
)
RETURNS TABLE(
    employee_id INTEGER,
    full_name   TEXT,
    department  TEXT,
    salary      NUMERIC
)
LANGUAGE plpgsql
STABLE
AS $$
BEGIN
    -- RETURN QUERY 可以调用多次，结果会合并
    RETURN QUERY
        SELECT e.employee_id,
               e.first_name || ' ' || e.last_name,
               d.department_name,
               e.salary
        FROM employees e
        LEFT JOIN departments d ON e.department_id = d.department_id
        WHERE (p_keyword IS NULL
               OR e.first_name ILIKE '%' || p_keyword || '%'
               OR e.last_name ILIKE '%' || p_keyword || '%')
          AND (p_dept_id IS NULL OR e.department_id = p_dept_id)
          AND (p_min_sal IS NULL OR e.salary >= p_min_sal)
        ORDER BY e.salary DESC;

    -- 检查是否有结果
    IF NOT FOUND THEN
        RAISE NOTICE '没有找到匹配的员工';
    END IF;
END;
$$;

-- 多种调用方式
SELECT * FROM search_employees('张');
SELECT * FROM search_employees(p_dept_id => 10, p_min_sal => 5000);
```

### 示例 7：PERFORM 执行无返回值语句

```sql
-- 使用 PERFORM 调用函数但不需要返回值
CREATE OR REPLACE FUNCTION cleanup_expired_sessions()
RETURNS INTEGER
LANGUAGE plpgsql
VOLATILE
AS $$
DECLARE
    v_deleted INTEGER;
BEGIN
    -- 删除过期会话
    DELETE FROM user_sessions
    WHERE expire_at < NOW();
    GET DIAGNOSTICS v_deleted = ROW_COUNT;

    -- PERFORM 调用函数，丢弃返回值
    -- 如果用 SELECT 会报错（结果无处存放）
    PERFORM pg_notify('session_cleanup',
            format('已清理 %s 个过期会话', v_deleted));

    -- 记录日志（假设 log_action 是另一个函数）
    PERFORM log_action('cleanup', format('删除 %s 条过期会话', v_deleted));

    RETURN v_deleted;
END;
$$;
```

### 示例 8：EXECUTE 动态 SQL

```sql
-- 动态表查询函数
CREATE OR REPLACE FUNCTION dynamic_count(
    p_table_name TEXT,
    p_where      TEXT DEFAULT NULL
)
RETURNS BIGINT
LANGUAGE plpgsql
VOLATILE
AS $$
DECLARE
    v_sql   TEXT;
    v_count BIGINT;
BEGIN
    -- 构建动态 SQL，使用 format 的 %I 防止 SQL 注入
    v_sql := format('SELECT count(*) FROM %I', p_table_name);

    IF p_where IS NOT NULL AND p_where != '' THEN
        v_sql := v_sql || ' WHERE ' || p_where;
    END IF;

    RAISE NOTICE '执行SQL: %', v_sql;

    -- EXECUTE 执行动态 SQL
    EXECUTE v_sql INTO v_count;

    RETURN v_count;
END;
$$;

SELECT dynamic_count('employees');
SELECT dynamic_count('employees', 'salary > 5000');

-- 使用 USING 绑定参数（更安全）
CREATE OR REPLACE FUNCTION dynamic_find_by_id(
    p_table_name TEXT,
    p_id_column  TEXT,
    p_id_value   INTEGER
)
RETURNS SETOF RECORD
LANGUAGE plpgsql
VOLATILE
AS $$
DECLARE
    v_sql TEXT;
BEGIN
    -- %I 用于标识符（表名/列名），$1 用于值绑定
    v_sql := format('SELECT * FROM %I WHERE %I = $1', p_table_name, p_id_column);

    -- USING 将参数值安全地绑定到 $1
    RETURN QUERY EXECUTE v_sql USING p_id_value;
END;
$$;
```

## 常见问题

### Q1：异常处理会影响性能吗？

是的。PostgreSQL 在进入包含 EXCEPTION 子句的 BEGIN 块时会创建一个内部 SAVEPOINT，这有一定开销。如果异常不发生，开销很小；但如果在循环中使用异常处理，性能影响会累积。

**建议：**
- 不要在循环内部使用 EXCEPTION（将整个循环放在一个 BEGIN...EXCEPTION 块中）
- 对于可以预判的错误，优先使用 IF 检查而非异常捕获
- 仅在确实需要时才使用异常处理

### Q2：游标和 FOR 循环遍历查询结果有什么区别？

`FOR rec IN SELECT ...` 是对游标的一种便捷封装，底层也使用游标。主要区别：

- **FOR 循环**：自动管理游标的打开和关闭，语法更简洁，推荐优先使用
- **显式游标**：可以更灵活地控制读取（如跳过行、滚动），可以传递给其他函数，适合需要精细控制的场景

### Q3：EXECUTE 动态 SQL 如何防止 SQL 注入？

```sql
-- 错误：直接拼接用户输入（危险！）
EXECUTE 'SELECT * FROM users WHERE name = ''' || p_name || '''';

-- 正确方式1：使用 USING 绑定参数值
EXECUTE 'SELECT * FROM users WHERE name = $1' USING p_name;

-- 正确方式2：使用 format 的 %L（字面量）和 %I（标识符）
EXECUTE format('SELECT * FROM %I WHERE name = %L', p_table, p_name);

-- %I 会自动加双引号并转义标识符
-- %L 会自动加单引号并转义字面量
```
