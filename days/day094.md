## 理论讲解

### 1. Web 应用数据库访问架构

现代 Web 应用与数据库的交互通常遵循分层架构：

```
┌─────────────┐    ┌─────────────┐    ┌────────────┐    ┌──────────┐
│   客户端     │ →  │  Web 服务器  │ →  │ 连接池/ORM  │ →  │ PostgreSQL│
│ Browser/App │    │ API Server  │    │ Driver     │    │ Database │
└─────────────┘    └─────────────┘    └────────────┘    └──────────┘
```

不同语言/框架选择数据库访问方式：

| 语言 | 驱动 | ORM/查询构建器 | 常用框架 |
|------|------|---------------|---------|
| Node.js | pg | Prisma, Drizzle, Knex | Express, Fastify, NestJS |
| Go | pgx, lib/pq | GORM, sqlx, sqlc | Gin, Echo, Fiber |
| Java | JDBC | MyBatis, JPA/Hibernate | Spring Boot |
| Python | psycopg2/3 | SQLAlchemy, Django ORM | Django, FastAPI, Flask |

### 2. 连接池的重要性

Web 应用并发处理大量请求，每个请求都创建和销毁数据库连接开销巨大。连接池预先创建一组连接，复用这些连接来服务请求。

关键参数：
- **最小连接数（min）**：池中始终保持的连接数
- **最大连接数（max）**：池能创建的最大连接数
- **空闲超时**：空闲连接多久后被回收
- **连接超时**：等待获取连接的最大时间

推荐的最大连接数公式：`max_connections = (CPU 核心数 * 2) + 有效磁盘数`

### 3. N+1 查询问题

这是 ORM 最常见的性能陷阱：查询 N 条主记录后，又为每条记录单独查询关联数据，总共执行 N+1 次查询。解决方案：预加载（Eager Loading）、JOIN 查询、DataLoader 模式。

### 4. REST API 数据库最佳实践

- 使用分页避免返回过多数据
- 参数化查询防止 SQL 注入
- 合理使用事务保证数据一致性
- 用数据库约束而非应用层逻辑保证数据完整性
- 实现乐观锁或悲观锁处理并发更新

### 示例1：Node.js pg 驱动

```javascript
// 安装: npm install pg

const { Pool } = require('pg');

// 创建连接池
const pool = new Pool({
  host: 'localhost',
  port: 5432,
  database: 'mydb',
  user: 'postgres',
  password: 'secret',
  max: 20,              // 最大连接数
  idleTimeoutMillis: 30000,
  connectionTimeoutMillis: 2000,
});

// 基本查询
async function getUsers() {
  const client = await pool.connect();
  try {
    const result = await client.query('SELECT * FROM users LIMIT 10');
    return result.rows;
  } finally {
    client.release(); // 归还连接到池
  }
}

// 参数化查询（防止 SQL 注入）
async function getUserById(id) {
  // $1, $2 是 PostgreSQL 的参数占位符
  const result = await pool.query(
    'SELECT * FROM users WHERE id = $1',
    [id]
  );
  return result.rows[0];
}

// 事务处理
async function transferFunds(fromId, toId, amount) {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    // 使用 FOR UPDATE 加行锁
    const from = await client.query(
      'SELECT balance FROM accounts WHERE id = $1 FOR UPDATE',
      [fromId]
    );

    if (from.rows[0].balance < amount) {
      throw new Error('余额不足');
    }

    await client.query(
      'UPDATE accounts SET balance = balance - $1 WHERE id = $2',
      [amount, fromId]
    );
    await client.query(
      'UPDATE accounts SET balance = balance + $1 WHERE id = $2',
      [amount, toId]
    );

    await client.query('COMMIT');
    return { success: true };
  } catch (e) {
    await client.query('ROLLBACK');
    throw e;
  } finally {
    client.release();
  }
}

// 批量插入
async function batchInsertUsers(users) {
  const values = users.map((u, i) =>
    `($${i*2+1}, $${i*2+2})`
  ).join(', ');

  const params = users.flatMap(u => [u.username, u.email]);

  const result = await pool.query(
    `INSERT INTO users (username, email) VALUES ${values} RETURNING id`,
    params
  );
  return result.rows;
}

// Express 路由示例
const express = require('express');
const app = express();
app.use(express.json());

// 分页查询
app.get('/api/users', async (req, res) => {
  const page = parseInt(req.query.page) || 1;
  const limit = Math.min(parseInt(req.query.limit) || 20, 100);
  const offset = (page - 1) * limit;

  const [data, count] = await Promise.all([
    pool.query(
      'SELECT id, username, email, created_at FROM users ORDER BY id LIMIT $1 OFFSET $2',
      [limit, offset]
    ),
    pool.query('SELECT COUNT(*) FROM users')
  ]);

  res.json({
    data: data.rows,
    pagination: {
      page,
      limit,
      total: parseInt(count.rows[0].count),
      pages: Math.ceil(count.rows[0].count / limit)
    }
  });
});
```

```javascript
        create: [
          { title: 'PostgreSQL入门', content: '...' , published: true },
          { title: '高级SQL技巧', content: '...' },
        ]
      }
    },
    include: { posts: true } // 预加载关联数据
  });

  // 查询：解决 N+1 问题
  const usersWithPosts = await prisma.user.findMany({
    include: {
      posts: {
        where: { published: true },
        orderBy: { createdAt: 'desc' },
        take: 5
      }
    }
  });

  // 复杂条件查询
  const results = await prisma.post.findMany({
    where: {
      AND: [
        { published: true },
        { author: { username: { contains: '张' } } },
        { tags: { some: { name: 'postgresql' } } }
      ]
    },
    select: {
      id: true,
      title: true,
      author: { select: { username: true } },
      _count: { select: { tags: true } }
    },
    orderBy: { createdAt: 'desc' },
    skip: 0,
    take: 20
  });

  // 事务
  const [updatedUser, newPost] = await prisma.$transaction([
    prisma.user.update({
      where: { id: 1 },
      data: { email: 'new@example.com' }
    }),
    prisma.post.create({
      data: { title: '新文章', authorId: 1 }
    })
  ]);

  // 聚合查询
  const stats = await prisma.post.aggregate({
    _count: { id: true },
    _avg: { authorId: true },
    where: { published: true }
  });

  // 分组统计
  const groupStats = await prisma.post.groupBy({
    by: ['authorId'],
    _count: { id: true },
    having: { id: { _count: { gt: 5 } } }
  });
}
```

### 示例3：Go pgx 驱动

```go
package main

import (
    "context"
    "fmt"
    "log"
    "time"

    "github.com/jackc/pgx/v5"
    "github.com/jackc/pgx/v5/pgxpool"
)

// 用户模型
type User struct {
    ID        int       `db:"id"`
    Username  string    `db:"username"`
    Email     string    `db:"email"`
    CreatedAt time.Time `db:"created_at"`
}

func main() {
    ctx := context.Background()

    // 创建连接池
    config, err := pgxpool.ParseConfig(
        "postgresql://postgres:secret@localhost:5432/mydb",
    )
    if err != nil {
        log.Fatal(err)
    }

    config.MaxConns = 20
    config.MinConns = 5
    config.MaxConnLifetime = 30 * time.Minute
    config.MaxConnIdleTime = 5 * time.Minute

    pool, err := pgxpool.NewWithConfig(ctx, config)
    if err != nil {
        log.Fatal(err)
    }
    defer pool.Close()

    // 单行查询
    var user User
    err = pool.QueryRow(ctx,
        "SELECT id, username, email, created_at FROM users WHERE id = $1", 1,
    ).Scan(&user.ID, &user.Username, &user.Email, &user.CreatedAt)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Printf("用户: %+v\n", user)

    // 多行查询
    rows, err := pool.Query(ctx,
        "SELECT id, username, email, created_at FROM users ORDER BY id LIMIT $1",
        10,
    )
    if err != nil {
        log.Fatal(err)
    }
    defer rows.Close()

    users, err := pgx.CollectRows(rows, pgx.RowToStructByName[User])
    if err != nil {
        log.Fatal(err)
    }
    for _, u := range users {
        fmt.Printf("%d: %s (%s)\n", u.ID, u.Username, u.Email)
    }

    // 事务
    tx, err := pool.Begin(ctx)
    if err != nil {
        log.Fatal(err)
    }
    // 使用 defer Rollback，如果已 Commit 则 Rollback 是空操作
    defer tx.Rollback(ctx)

    _, err = tx.Exec(ctx,
        "INSERT INTO users (username, email) VALUES ($1, $2)",
        "新用户", "new@example.com",
    )
    if err != nil {
        log.Fatal(err)
    }

    err = tx.Commit(ctx)
    if err != nil {
        log.Fatal(err)
    }

    // 批量操作（使用 pgx Batch）
    batch := &pgx.Batch{}
    batch.Queue("INSERT INTO users (username, email) VALUES ($1, $2)",
        "batch1", "b1@example.com")
    batch.Queue("INSERT INTO users (username, email) VALUES ($1, $2)",
        "batch2", "b2@example.com")
    batch.Queue("INSERT INTO users (username, email) VALUES ($1, $2)",
        "batch3", "b3@example.com")

    br := pool.SendBatch(ctx, batch)
    defer br.Close()

    for i := 0; i < batch.Len(); i++ {
        _, err := br.Exec()
        if err != nil {
            log.Printf("批量操作第%d条失败: %v", i+1, err)
        }
    }

    // COPY 协议高速导入
    copyCount, err := pool.CopyFrom(
        ctx,
        pgx.Identifier{"users"},
        []string{"username", "email"},
        pgx.CopyFromRows([][]interface{}{
            {"copy1", "c1@example.com"},
            {"copy2", "c2@example.com"},
            {"copy3", "c3@example.com"},
        }),
    )
    fmt.Printf("COPY 导入 %d 行\n", copyCount)
}
```

### 示例5：REST API 最佳实践

```javascript
// Express + pg 完整 REST API 示例
const express = require('express');
const { Pool } = require('pg');

const app = express();
app.use(express.json());

const pool = new Pool({
  connectionString: process.env.DATABASE_URL,
  max: 20
});

// 中间件：请求日志
app.use((req, res, next) => {
  const start = Date.now();
  res.on('finish', () => {
    console.log(`${req.method} ${req.path} ${res.statusCode} ${Date.now() - start}ms`);
  });
  next();
});

// GET /api/products - 列表查询（分页+搜索+排序）
app.get('/api/products', async (req, res) => {
  try {
    const {
      page = 1,
      limit = 20,
      search = '',
      sort = 'id',
      order = 'asc'
    } = req.query;

    // 白名单验证排序字段，防止 SQL 注入
    const allowedSorts = ['id', 'name', 'price', 'created_at'];
    const safeSort = allowedSorts.includes(sort) ? sort : 'id';
    const safeOrder = order === 'desc' ? 'DESC' : 'ASC';
    const safeLimit = Math.min(parseInt(limit) || 20, 100);
    const offset = (Math.max(parseInt(page) || 1, 1) - 1) * safeLimit;

    const queryText = `
      SELECT id, name, price, stock, created_at
      FROM products
      WHERE ($1 = '' OR name ILIKE '%' || $1 || '%')
      ORDER BY ${safeSort} ${safeOrder}
      LIMIT $2 OFFSET $3
    `;

    const countText = `
      SELECT COUNT(*) FROM products
      WHERE ($1 = '' OR name ILIKE '%' || $1 || '%')
    `;

    const [data, count] = await Promise.all([
      pool.query(queryText, [search, safeLimit, offset]),
      pool.query(countText, [search])
    ]);

    res.json({
      data: data.rows,
      meta: {
        total: parseInt(count.rows[0].count),
        page: parseInt(page),
        limit: safeLimit,
        pages: Math.ceil(count.rows[0].count / safeLimit)
      }
    });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: '服务器内部错误' });
  }
});

// POST /api/products - 创建（带验证）
app.post('/api/products', async (req, res) => {
  const { name, price, stock } = req.body;

  if (!name || price == null || price < 0) {
    return res.status(400).json({ error: '参数校验失败' });
  }

  try {
    const result = await pool.query(
      `INSERT INTO products (name, price, stock)
       VALUES ($1, $2, $3) RETURNING *`,
      [name, price, stock || 0]
    );
    res.status(201).json(result.rows[0]);
  } catch (err) {
    if (err.code === '23505') { // unique_violation
      return res.status(409).json({ error: '产品名称已存在' });
    }
    console.error(err);
    res.status(500).json({ error: '服务器内部错误' });
  }
});

// PUT /api/products/:id - 更新（乐观锁）
app.put('/api/products/:id', async (req, res) => {
  const { name, price, stock, version } = req.body;

  try {
    const result = await pool.query(
      `UPDATE products
       SET name = COALESCE($1, name),
           price = COALESCE($2, price),
           stock = COALESCE($3, stock),
           version = version + 1,
           updated_at = NOW()
       WHERE id = $4 AND version = $5
       RETURNING *`,
      [name, price, stock, req.params.id, version]
    );

    if (result.rowCount === 0) {
      return res.status(409).json({ error: '数据已被其他用户修改，请刷新重试' });
    }
    res.json(result.rows[0]);
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: '服务器内部错误' });
  }
});

app.listen(3000, () => console.log('Server running on :3000'));
```

## 常见问题

### Q1: Web 应用中数据库连接池大小应该设置多少？

**A:** 一个常见的误区是连接数越多越好。实际上过多的连接会增加 PostgreSQL 进程调度开销。推荐公式：`连接数 = (CPU 核心数 * 2) + 有效磁盘数`。例如 4 核服务器建议设置 9-10 个连接。如果应用有多个实例，总连接数应在 PostgreSQL 的 `max_connections` 范围内。可以使用 PgBouncer 作为外部连接池来支持更多应用实例。

### Q2: 如何防止 Web 应用中的 SQL 注入？

**A:** 核心原则是**永远使用参数化查询**，不要拼接 SQL 字符串。所有主流驱动都支持参数化：pg 用 `$1, $2`，psycopg2 用 `%s`，JDBC 用 `?`，pgx 用 `$1`。对于动态排序字段等无法参数化的部分，使用白名单验证。ORM 框架（Prisma、SQLAlchemy、JPA）默认就是参数化的。

### Q3: 选 ORM 还是原生 SQL？

**A:** 取决于项目需求。ORM 适合 CRUD 密集的业务应用，提高开发效率和可维护性。原生 SQL 适合复杂分析查询、性能敏感场景。许多项目混合使用：简单操作用 ORM，复杂查询用原生 SQL。像 MyBatis 和 Prisma 的 `$queryRaw` 就是这种混合模式的代表。
